func compareTokenWithExpected(t *testing.T, tokenStr string, expectedPath string) {
	t.Helper()

	// 1. Decode token payload
	parts := strings.Split(tokenStr, ".")
	if len(parts) != 3 {
		t.Fatalf("invalid token format: expected 3 parts, got %d", len(parts))
	}

	payloadJSON, err := base64.RawURLEncoding.DecodeString(parts[1])
	if err != nil {
		t.Fatalf("failed to decode payload: %v", err)
	}

	var actual map[string]interface{}
	if err := json.Unmarshal(payloadJSON, &actual); err != nil {
		t.Fatalf("failed to unmarshal actual payload: %v", err)
	}

	// 2. Load expected
	expectedData, err := os.ReadFile(expectedPath)
	if err != nil {
		t.Fatalf("failed to read expected file %s: %v", expectedPath, err)
	}

	var expected map[string]interface{}
	if err := json.Unmarshal(expectedData, &expected); err != nil {
		t.Fatalf("failed to unmarshal expected payload: %v", err)
	}

	// 3. Handle placeholders and compare
	for key, expectedVal := range expected {
		actualVal, ok := actual[key]
		if !ok {
			t.Errorf("missing claim: %s", key)
			continue
		}

		if expectedVal == "<DYNAMIC>" {
			continue // skip comparison for dynamic fields
		}

		if reflect.DeepEqual(expectedVal, actualVal) {
			continue
		}

		// Use json.Marshal for deep comparison of arrays/objects to handle types correctly
		expectedBytes, _ := json.Marshal(expectedVal)
		actualBytes, _ := json.Marshal(actualVal)

		if string(expectedBytes) != string(actualBytes) {
			t.Errorf("claim mismatch for %s:\nexpected: %v\nactual:   %v", key, string(expectedBytes), string(actualBytes))
		}
	}

	// Check for unexpected extra claims
	for key := range actual {
		if _, ok := expected[key]; !ok {
			t.Errorf("unexpected extra claim: %s (value: %v)", key, actual[key])
		}
	}
}
